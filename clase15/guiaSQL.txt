para CREAR usamos 

CREATE TABLE {nombre de la base + nombre de tabla}
segundo paso va a ser la descripcion de los campos (columnas)
{key + tipo + atributos }
(`id` INT NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(20) NOT NULL DEFAULT 'sin nombre' , `edad` INT NOT NULL , `descripcion` TEXT NOT NULL , PRIMARY KEY (`id`))
ENGINE = InnoDB;


PARA AGREGAR UN REGISTRO 
INSERT INTO `usuario`(`nombre`, `edad`, `descripcion`) VALUES ('jose',27,'Me llamo jose y soy de mar del plata')
UTILIZAMOS INTRO PARA DEFINIR UNA INSERCION 
DESPUES EL NOMBRE DE LA TABLA
LUEGO PONEMOS LOS CAMPOS A AGREGAR 
Y SEPARA POR VALUE LOS VALORES ORDENADOS A INSERTAR
EN RESULTADO VA A INSERTAR: 
TABLA USARIOS 
    nombre:'jose',
    edad:27,
    descripcion:'Me llamo jose y soy de mar del plata'


UPDATE
UPDATE `usuario` SET `id`='[value-1]',`nombre`='[value-2]',`edad`='[value-3]',`descripcion`='[value-4]' WHERE 1

UTILIZAMOS UPDATE PARA ACTULIZAR 
DONDE SE LE AGREGAR EL NOMBRE DE LA TABLA Y LUEGO 
LA PALABRA SET VA A HACER REFERENCIA A LO QUE VAMOS A ACTUALIZAR POR EJEMPLO SI QUERES ACTUALIZAR LA EDAD 
TENDRIAMOS QUE PONER `edad`=30
Y LUEGO CON LA PALABRA WHERE ( NO ES OBLIGATORIA PERO DEBERIA )
DEFINIMOS A QUIEN VAMOS A ACTUALIZAR POR EJEMPLO 
`id`=9


PARA ELIMIAR USAMOS 
DELETE 
DELETE FROM `usuario` WHERE `nombre`='JUAN' AND `edad` > 30

FROM DESCRIBE LA TABLA
Y WHERE LA CONSULTA SOBRE QUIEN VAMOS A ELIMINAR
